\chapter{Triển khai kỹ thuật}
\label{ch:implementation}

% Giới thiệu ngắn
Chương này trình bày chi tiết về khía cạnh công nghệ phần mềm và các bước triển khai mã nguồn thực tiễn của kiến trúc SENTINEL. Thay vì lý thuyết trừu tượng, nội dung ở đây tập trung vào sơ đồ tổ chức thư mục, các lớp đối tượng cốt lõi (Classes/Methods), các dịch vụ Docker và thiết kế Giao diện quản trị đồ họa (Streamlit Dashboard) hỗ trợ tương tác giữa con người và máy (Human-in-the-Loop).

\section{Môi trường Công nghệ và Cấu trúc Mã nguồn Dự án}
\begin{itemize}
    \item \textbf{Công nghệ cốt lõi:} Python 3.10+, LangGraph, LangChain, FAISS, SQLite, Streamlit, Docker, llama.cpp (với thư viện Llama-cpp-python).
    \item \textbf{Cấu trúc thư mục (Directory Structure):} Biểu diễn cây thư mục (Tree) chi tiết, phân chia rạch ròi các thành phần: \texttt{src/core/} (Tier 1 \& 2), \texttt{src/rag/} (Vector DB), \texttt{src/security/} (Guardrails), \texttt{src/dashboard/} (UI), \texttt{experiments/}, \texttt{tests/}.
\end{itemize}

\section{Triển khai Tier-1: Động cơ Lọc \& Welford}
\begin{itemize}
    \item \textbf{Lớp \texttt{RunningStats} và \texttt{SessionBaseline}:} Triển khai mã nguồn Python cho thuật toán Welford. Giải quyết bài toán cấp phát bộ nhớ từ điển (dictionary allocation) bằng hàm hash IP và dọn dẹp bộ đệm (TTL cache).
    \item \textbf{Lớp \texttt{RuleEngine}:} Thiết kế pattern Matching dựa trên Regex và Static Thresholds.
\end{itemize}

\section{Triển khai Tier-2: AI Agent và RAG}
\begin{itemize}
    \item \textbf{Lớp \texttt{OpenAILLMClient} / Llama.cpp Wrapper:} Cách thiết lập thông số cho Local LLM (\texttt{temperature=0.0} để tối ưu tính xác định, cấu hình context size, \texttt{n\_gpu\_layers}).
    \item \textbf{Thiết lập LangGraph Workflow:} Cách sử dụng \texttt{StateGraph} để định nghĩa các node \texttt{analyze\_node}, \texttt{retrieve\_node} và hàm \texttt{should\_retrieve} cho các cạnh điều kiện.
    \item \textbf{Lớp \texttt{DualRetriever}:} Triển khai FAISS Index và rank\_bm25 cho tính năng hợp nhất Reciprocal Rank Fusion (RRF). Mã nguồn mô tả thuật toán chấm điểm và sắp xếp (sorting algorithm).
\end{itemize}

\section{Cài đặt Hệ thống Bảo vệ Guardrail và HMAC}
\begin{itemize}
    \item \textbf{Lớp \texttt{DelimitedDataEncapsulator}:} Mã sinh chuỗi token hệ thập lục phân ngẫu nhiên thông qua \texttt{os.urandom} và \texttt{hashlib} để bọc payload logs.
    \item \textbf{Lớp \texttt{HMACHashing} (Database):} Đặc tả sơ đồ lược đồ CSDL SQLite (\texttt{threat\_memory.db}), trigger tính toán hàm băm SHA-256 từ Previous Hash + Current Data.
\end{itemize}

\section{Triển khai Giao diện Quản trị SOC (Streamlit Dashboard)}
\begin{itemize}
    \item \textbf{Kiến trúc Dashboard:} Ứng dụng mô hình Frontend (Streamlit) giao tiếp với Backend (SQLite/File logs) qua các thành phần giao diện động.
    \item \textbf{Các mô-đun Giao diện hiển thị:}
        \begin{itemize}
            \item \textit{Trang Tổng quan (Real-time Queue):} Trực quan hóa số liệu cảnh báo Tier-1 và phán quyết Tier-2.
            \item \textit{Trang Phân tích Đe dọa (Threat Memory):} Hiển thị đồ thị cảnh báo APT theo dải thời gian IP.
            \item \textit{Trang Kiểm toán Toàn vẹn (Audit Trail):} Module tự động xác minh HMAC Log Chaining bằng giao diện đèn giao thông (Xanh = Hợp lệ, Đỏ = Đứt gãy).
        \end{itemize}
    \item \textbf{Human-in-the-loop (HITL):} Giao diện duyệt (Approve/Reject) các bản án được AI đề xuất để cập nhật thành Firewall Rules thực thụ.
\end{itemize}

\section{Đóng gói và Triển khai Hạ tầng (Containerization)}
\begin{itemize}
    \item \textbf{Đặc tả \texttt{Dockerfile}:} Cơ chế đa giai đoạn (multi-stage build) để giảm dung lượng file ảnh, biên dịch thư viện C++ (llama.cpp) cùng hỗ trợ CUDA.
    \item \textbf{Quản lý Dịch vụ \texttt{docker-compose.yml}:} Liên kết container của Dashboard, Agent Engine và volume gắn kết tập tin Logs chia sẻ.
\end{itemize}

Tóm lại, Chương 4 đã hệ thống hóa quá trình lập trình phần mềm SENTINEL từ các thuật toán lõi đến giao diện người dùng và cơ sở hạ tầng ảo hóa. Chương 5 tiếp theo sẽ thiết lập bài toán kiểm thử để đánh giá định lượng các thành phần này.
