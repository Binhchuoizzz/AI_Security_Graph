\chapter{Kết luận và Hướng phát triển}
\label{ch:conclusion}

% Giới thiệu ngắn
Chương cuối cùng này đúc kết lại toàn bộ hành trình nghiên cứu của đề tài "Kiến trúc nhận thức hai tầng cho phát hiện và phản hồi mối đe dọa tự động sử dụng AI Tác tử" (SENTINEL). Chương tổng hợp các kết quả khoa học đã được chứng minh, tự nhận xét những mặt còn hạn chế trong khuôn khổ luận văn thạc sĩ và mở ra các chân trời nghiên cứu mới cho ngành an ninh mạng kết hợp Trí tuệ Nhân tạo.

\section{Tổng hợp các Kết quả Nghiên cứu Đạt được}
\begin{itemize}
    \item \textbf{Trả lời các câu hỏi nghiên cứu:} Khẳng định lại việc giải quyết thành công bài toán độ trễ và Alert Fatigue thông qua kiến trúc phân lớp Tier-1 Welford và Tier-2 LangGraph Agent.
    \item \textbf{Bảo mật hệ thống AI (AI Security):} Chứng minh tính khả thi của kỹ thuật Delimited Data Encapsulation và cấu trúc HMAC Chaining trong việc phòng chống các rủi ro thao túng hệ thống nhận thức nội bộ của doanh nghiệp (Prompt Injection, Data Tampering).
    \item \textbf{Giải quyết bài toán Zero-day \& APT:} Sự kết hợp xuất sắc giữa kỹ thuật theo dõi chuỗi cảnh báo lịch sử (Threat Memory) và kho dữ liệu lai (Dual-RAG MITRE + NIST) mang đến nhận thức ngữ cảnh sâu rộng, giúp hệ thống phơi bày các chiến dịch tinh vi nhất.
    \item \textbf{Kiểm định tính đúng đắn:} Các thành quả trên không dừng ở mặt lý thuyết mà được xác thực mạnh mẽ bởi các độ đo 5D và kiểm định thống kê chuẩn mực (McNemar, Mann-Whitney U) trên dữ liệu thực tế.
\end{itemize}

\section{Đóng góp Khoa học và Thực tiễn}
\begin{itemize}
    \item \textbf{Giá trị khoa học:} Cung cấp một khung kiến trúc tiêu chuẩn (framework) có tính module hóa cao về việc kết hợp Thống kê Toán học và Mô hình Ngôn ngữ Lớn trong an ninh mạng. Củng cố kho tàng lý thuyết về bảo mật LLM.
    \item \textbf{Giá trị thực tiễn:} Sản phẩm hoàn thiện SENTINEL có thể được sử dụng làm công cụ lõi hoặc công cụ hỗ trợ độc lập cho các doanh nghiệp vừa và nhỏ (SMBs) không đủ ngân sách vận hành đội ngũ SOC Tier-1 quy mô lớn, mà vẫn đảm bảo tính bảo mật Zero Trust khi vận hành offline.
\end{itemize}

\section{Những Hạn chế còn Tồn đọng}
Mặc dù hệ thống thể hiện sự ưu việt, nghiên cứu vẫn còn những rào cản nhất định:
\begin{itemize}
    \item \textbf{Định dạng log đầu vào:} Hệ thống hiện thử nghiệm sâu trên lưu lượng mạng (Network Flow logs) và HTTP payload. Căn cứ thực tế đòi hỏi xử lý thêm rất nhiều định dạng log phi cấu trúc khác (Windows Event Logs, Linux Syslogs, CloudTrail) dẫn đến yêu cầu tinh chỉnh parser lớn hơn.
    \item \textbf{Rào cản Phần cứng:} Tốc độ suy luận dù đã được tăng tốc bằng bộ đệm (Semantic Cache) nhưng khi gặp cảnh báo hoàn toàn mới, LLM vẫn cần tốn thời gian suy luận, đòi hỏi ngân sách đầu tư GPU phần cứng chuyên dụng khá cao nếu quy mô luồng mạng hàng trăm Gigabit/s.
    \item \textbf{Thiết kế Mitigation Vật lý:} Nghiên cứu cố tình đóng khung hệ thống ở mức đề xuất (Recommend Firewall Rules) thay vì can thiệp vật lý vào router để tránh tấn công tự từ chối dịch vụ (Self-DoS), tuy nhiên điều này hạn chế khả năng Real-time Blocking thực thụ.
\end{itemize}

\section{Định hướng Phát triển Tương lai}
Từ những hạn chế trên, hướng đi mở rộng cho hệ thống bao gồm:
\begin{itemize}
    \item \textbf{Học liên kết (Federated Learning):} Mở rộng SENTINEL thành kiến trúc phi tập trung. Thay vì chia sẻ logs mạng (vi phạm quyền riêng tư), các điểm cuối SOC chỉ chia sẻ tham số mô hình đã học được về các biến thể mã độc mới cho nhau.
    \item \textbf{Kiến trúc Đa Tác tử (Multi-Agent System):} Phân rã LangGraph Agent thành nhiều tác tử chuyên biệt: Malware Analyst Agent, Network Analyst Agent, Threat Intel Agent tranh luận với nhau trước khi đưa ra bản án chung thẩm, nâng cao hơn nữa độ chính xác.
    \item \textbf{Kiểm thử Tự động (Automated Red Teaming):} Xây dựng một tác tử Red Team nội bộ liên tục sinh ra các payload tấn công tinh vi nhắm vào Sentinel để tự đánh giá và nâng cấp khả năng phòng thủ của hệ thống.
\end{itemize}
